const handleVerifyOtp = async () => {
  try {
    const { data: authData, error: verifyError } = await supabase.auth.verifyOtp({
      email: email, token: otp, type: 'signup'
    })
    if (verifyError) { alert('OTP FAIL: ' + verifyError.message); return }
    if (!authData.user) { alert('OTP FAIL: No user returned'); return }
    
    alert('OTP Verified. Creating profile...')
    
    const savedReferral = localStorage.getItem('pending_referral_code')
    
    const { error: profileError } = await supabase.from('profiles').insert({
      user_id: authData.user.id,
      email: email,
      name: name,
      phone: phone,
      referral_code: 'FAIS' + Math.floor(1000 + Math.random() * 9000),
      referred_by_code: savedReferral || null,
      created_at: new Date().toISOString()
    })
    
    if (profileError) { 
      alert('DB ERROR: ' + profileError.message)
      return 
    }
    
    localStorage.removeItem('pending_referral_code')
    alert('SUCCESS: Account created!')
    
  } catch (err) {
    alert('VERIFY ERROR: ' + err.message)
  }
}
