import { useState, useEffect } from "react"
import { useLocation } from "wouter"
import { supabase } from "../lib/supabaseClient"

export default function LoginPage() {
  const [location] = useLocation()
  const refCode = new URLSearchParams(location.split('?')[1]).get('ref') || null
  const [email, setEmail] = useState("")
  const [pw, setPw] = useState("")
  const [otpCode, setOtpCode] = useState("")
  const [step, setStep] = useState("login")
  const [loading, setLoading] = useState(false)
  const [loginErr, setLoginErr] = useState("")
  const [otpErr, setOtpErr] = useState("")
  const [fullName, setFullName] = useState("")
  const [phone, setPhone] = useState("")

  // AUTO REDIRECT IF ALREADY LOGGED IN
  useEffect(() => {
    supabase.auth.getSession().then(({ data: { session } }) => {
      if (session) window.location.replace('/showcase')
    })

    const { data: { subscription } } = supabase.auth.onAuthStateChange((event, session) => {
      if (event === 'SIGNED_IN' && session) window.location.replace('/showcase')
    })
    return () => subscription.unsubscribe()
  }, [])

  const handleSignUp = async () => {
    setLoading(true)
    setOtpErr("")
    const { error } = await supabase.auth.signUp({
      email: email.trim().toLowerCase(),
      password: pw,
      options: { 
        data: { 
          full_name: fullName, 
          phone: phone, 
          referral_code: refCode 
        } 
      }
    })
    setLoading(false)
    if (error) {
      setOtpErr(error.message)
    } else {
      setStep('otp')
    }
  }

  const handleVerifyOtp = async () => {
    setLoading(true)
    setOtpErr("")
    const { data, error } = await supabase.auth.verifyOtp({
      email: email.trim().toLowerCase(),
      token: otpCode,
      type: 'signup'
    })
    setLoading(false)
    if (error) {
      setOtpErr('Invalid ya expired OTP. Dobara try karo.')
      return
    }
    if (data.session) {
      localStorage.clear()
      sessionStorage.clear()
      window.location.replace('/showcase')
    }
  }

  const handleLogin = async () => {
    setLoading(true)
    setLoginErr("")
    const { data, error } = await supabase.auth.signInWithPassword({
      email: email.trim().toLowerCase(),
      password: pw
    })
    setLoading(false)
    if (error) {
      setLoginErr('Email ya password galat hai')
      return
    }
    if (data.session) {
      localStorage.clear()
      sessionStorage.clear()
      window.location.replace('/showcase')
    }
  }

  // JSX same as old - UI change nahi karna
  return (
    //... purana UI code same rahega
  )
}
