  onClick={async () => {

  }}

FILE-3: Dashboard.js
OLD: useEffect(() => { const token = localStorage.getItem("token"); if (!token) setLocation("/login"); }, []);

  supabase.auth.getSession().then(({ data: { session } }) => {
    if (!session) setLocation("/login")
  })

FILE-5: AdminPanel.jsx
OLD: isAdminLoggedIn(), adminLogin, adminLogout, getAllUsers from "../lib/api"
NEW:
- Auth check: useEffect(() => {
    supabase.auth.getSession().then(async ({ data: { session } }) => {
      if (!session) { setAuthed(false); return }
      const { data: { user } } = await supabase.auth.getUser()
      const { data: profile } = await supabase.from('users').select('role').eq('id', user.id).single()
      if (profile?.role !== 'admin') setLocation('/dashboard')
      else setAuthed(true)
    })
  }, [])
- Admin login: supabase.auth.signInWithPassword() + role check from 'users' table
- Logout: supabase.auth.signOut()
- Data calls: getAllUsers() → supabase.from('users').select('*')
- approveWithdrawal, rejectDeposit etc → Supabase table update calls
- localStorage.isAdminLoggedIn() → DELETE
