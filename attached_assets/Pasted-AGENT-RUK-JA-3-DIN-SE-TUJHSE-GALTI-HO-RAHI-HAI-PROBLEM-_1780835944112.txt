import { useEffect, useState } from 'react'
import { supabase } from './supabaseClient' // apna path daal

export default function Profile() {
  const [profile, setProfile] = useState(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState(null)

  useEffect(() => {
    const getProfile = async () => {
      setLoading(true)
      const { data: { user } } = await supabase.auth.getUser()
      
      if (!user) {
        setError('Not logged in')
        setLoading(false)
        return
      }

      const { data, error } = await supabase
        .from('profiles')
        .select('name, phone, referral_code, referred_by_code')
        .eq('user_id', user.id)
        .maybeSingle()

      if (error) {
        setError(error.message)
      } else {
        setProfile(data)
      }
      setLoading(false)
    }
    getProfile()
  }, [])

  if (loading) return <div>Loading...</div>
  if (error) return <div>Error: {error}</div>

  return (
    <div>
      <h1>Profile</h1>
      <p>Name: {profile?.name || 'Not set'}</p>
      <p>Phone: {profile?.phone || 'Not set'}</p>
      <p>Your Referral Code: {profile?.referral_code || 'Not set'}</p>
      <p>Referred By: {profile?.referred_by_code || 'None'}</p>
    </div>
  )
}

STEP 5: TRIGGER CHECK KAR
SQL me ye chala:
SELECT * FROM information_schema.triggers WHERE event_object_table = 'users';

Agar trigger nahi hai to ye chala:
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
  INSERT INTO public.profiles (user_id, email, name, phone, referral_code, referred_by_code)
  VALUES (
    NEW.id,
    NEW.email,
    NEW.raw_user_meta_data->>'name',
    NEW.raw_user_meta_data->>'phone',
    UPPER(SUBSTRING(MD5(RANDOM()::TEXT), 1, 8)),
    NEW.raw_user_meta_data->>'referred_by_code'
  );
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();
