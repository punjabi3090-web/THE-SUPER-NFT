APPROVE FUNCTION:
function approveWithdrawal(requestId) {
  let requests = JSON.parse(localStorage.withdrawalRequests);
  let req = requests.find(r => r.id === requestId);
  let users = JSON.parse(localStorage.allUsers);
  let user = users.find(u => u.userId === req.userId);

  if(user.walletBalance < req.amount) return alert('Insufficient balance');

  user.walletBalance -= req.amount;
  user.totalWithdraw = (user.totalWithdraw || 0) + req.amount;
  req.status = 'approved';
  req.approvedDate = new Date().toISOString();
  req.approvedBy = 'admin@demo.com';

  localStorage.setItem('allUsers', JSON.stringify(users));
  localStorage.setItem('withdrawalRequests', JSON.stringify(requests));

  user.myActivityHistory.unshift({
    id: Date.now(), type: 'withdrawal', title: 'Withdrawal Approved',
    amount: req.amount, status: 'Completed', address: req.address,
    date: new Date().toLocaleString(), icon: '💸', color: '#EA4335'
  });

  logAdminAction('Approved withdrawal', req.userId, req.amount);
}

REJECT FUNCTION:
function rejectWithdrawal(requestId, reason) {
  let requests = JSON.parse(localStorage.withdrawalRequests);
  let req = requests.find(r => r.id === requestId);
  req.status = 'rejected';
  req.rejectReason = reason;
  req.rejectedDate = new Date().toISOString();
  localStorage.setItem('withdrawalRequests', JSON.stringify(requests));
  logAdminAction('Rejected withdrawal', req.userId, req.amount, reason);
}

AUTO REWARD SYSTEM

function checkAutoRewards(userId, depositAmount) {
  let settings = JSON.parse(localStorage.rewardSettings);
  let users = JSON.parse(localStorage.allUsers);
  let user = users.find(u => u.userId === userId);

  // 1. Deposit Bonus
  if(settings.depositBonus.enabled && depositAmount >= settings.depositBonus.minDeposit) {
    let bonus = Math.min(depositAmount * (settings.depositBonus.percentage / 100), settings.depositBonus.maxBonus);
    user.walletBalance += bonus;
    user.myActivityHistory.unshift({
      id: Date.now(), type: 'reward', title: 'Deposit Bonus',
      amount: bonus, date: new Date().toLocaleString(), icon: '🎁', color: '#FBBC04'
    });
  }

  // 2. Referral Bonus
  if(user.referredBy && settings.referralBonus.enabled) {
    let upline = users.find(u => u.referralCode === user.referredBy);
    if(upline) {
      let refBonus = depositAmount * (settings.referralBonus.level1 / 100);
      upline.walletBalance += refBonus;
      upline.myActivityHistory.unshift({
        id: Date.now(), type: 'commission', title: 'Referral Bonus L1',
        amount: refBonus, from: user.name, date: new Date().toLocaleString(), icon: '👥', color: '#9C27B0'
      });
    }
  }

  // 3. Level Upgrade
  let totalDep = user.totalDeposit;
  let newLevel = 1;
  if(totalDep >= settings.levelSystem.level4Deposit) newLevel = 4;
  else if(totalDep >= settings.levelSystem.level3Deposit) newLevel = 3;
  else if(totalDep >= settings.levelSystem.level2Deposit) newLevel = 2;

  if(newLevel > user.level) {
    let levelBonus = settings.levelSystem[`level${newLevel}Bonus`];
    user.level = newLevel;
    user.walletBalance += levelBonus;
    user.myActivityHistory.unshift({
      id: Date.now(), type: 'reward', title: `Level ${newLevel} Achieved`,
      amount: levelBonus, date: new Date().toLocaleString(), icon: '⭐', color: '#FBBC04'
    });
  }

  localStorage.setItem('allUsers', JSON.stringify(users));
}

HELPER FUNCTIONS

function getCurrentUser() {
  let users = JSON.parse(localStorage.allUsers || '[]');
  return users.find(u => u.userId === localStorage.currentUser);
}

function addToHistory(userId, type, data) {
  let users = JSON.parse(localStorage.allUsers);
  let user = users.find(u => u.userId === userId);
  if(!user.myActivityHistory) user.myActivityHistory = [];
  user.myActivityHistory.unshift({id: Date.now(), type,...data});
  localStorage.setItem('allUsers', JSON.stringify(users));
}

function logAdminAction(action, target, amount, details) {
  let logs = JSON.parse(localStorage.adminLogs || '[]');
  logs.unshift({
    id: Date.now(), admin: 'admin@demo.com', action, target,
    amount: amount || 0, details: details || '', timestamp: new Date().toISOString()
  });
  localStorage.setItem('adminLogs', JSON.stringify(logs));
}
