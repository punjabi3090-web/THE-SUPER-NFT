const loadTeamStats = async () => {
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return;

  const { data: myProfile } = await supabase
    .from('profiles')
    .select('referral_code')
    .eq('id', user.id)
    .single();
  
  const myCode = myProfile?.referral_code;
  if (!myCode) return;

  // LEVEL 1 - FIX: Add referral_code to select
  const { data: level1Users } = await supabase
    .from('profiles')
    .select('id, email, referral_code')  // ← THIS LINE WAS WRONG
    .eq('referred_by_code', myCode);

  const level1Count = level1Users?.length || 0;
  const level1Codes = level1Users?.map(u => u.referral_code) || [];
  const level1Ids = level1Users?.map(u => u.id) || [];

  // LEVEL 2 - This will work now
  let level2Users = [];
  if (level1Codes.length > 0) {
    const { data: l2Users } = await supabase
      .from('profiles')
      .select('id, email')
      .in('referred_by_code', level1Codes);
    level2Users = l2Users || [];
  }
  const level2Count = level2Users.length;
  const level2Ids = level2Users.map(u => u.id);

  // A ENTHUSIAST - Level 1 with deposit
  let aEnthusiastList = [];
  if (level1Ids.length > 0) {
    const { data: aDeps } = await supabase
      .from('deposits')
      .select('user_id')
      .in('user_id', level1Ids)
      .eq('status', 'completed')
      .gt('amount', 0);
    const aUserIds = [...new Set(aDeps?.map(d => d.user_id) || [])];
    aEnthusiastList = level1Users?.filter(u => aUserIds.includes(u.id)) || [];
  }

  // B/C ENTHUSIAST - Level 2 with deposit
  let bcEnthusiastList = [];
  if (level2Ids.length > 0) {
    const { data: bcDeps } = await supabase
      .from('deposits')
      .select('user_id')
      .in('user_id', level2Ids)
      .eq('status', 'completed')
      .gt('amount', 0);
    const bcUserIds = [...new Set(bcDeps?.map(d => d.user_id) || [])];
    bcEnthusiastList = level2Users?.filter(u => bcUserIds.includes(u.id)) || [];
  }

  setTeamStats({
    totalRegister: level1Count,
    bcRegister: level2Count,  // This goes in "Total B/C Register Members"
    aEnthusiast: aEnthusiastList.length,
    bcEnthusiast: bcEnthusiastList.length
  });

  // For click modals
  setLevel1List(level1Users || []);
  setLevel2List(level2Users || []);
  setAEnthusiastList(aEnthusiastList);
  setBcEnthusiastList(bcEnthusiastList);
}
