CREATE OR REPLACE FUNCTION get_user_dashboard(p_user_id UUID, p_referral_code TEXT)
RETURNS JSON AS $$
DECLARE
  level1_ids UUID[];
  level2_ids UUID[];
BEGIN
  -- Level 1 = Direct referrals = A enthusiasts
  SELECT ARRAY(SELECT user_id FROM profiles WHERE referred_by_code = p_referral_code) INTO level1_ids;
  
  -- Level 2 = B enthusiasts  
  SELECT ARRAY(SELECT user_id FROM profiles WHERE referred_by_code = ANY(
    SELECT referral_code FROM profiles WHERE user_id = ANY(level1_ids)
  )) INTO level2_ids;

  RETURN json_build_object(
    'teamCount', (SELECT COUNT(*) FROM profiles WHERE referred_by_code = p_referral_code),
    'totalRegisterMembers', (SELECT COUNT(*) FROM profiles WHERE referred_by_code = p_referral_code),
    'aEnthusiasts', array_length(level1_ids, 1),
    'validMembers', (
      SELECT COUNT(DISTINCT p.user_id) FROM profiles p 
      JOIN deposits d ON d.user_id = p.user_id 
      WHERE p.user_id = ANY(level1_ids) AND d.status = 'completed' AND d.amount > 0
    ),
    'bEnthusiasts', (
      SELECT COUNT(DISTINCT p.user_id) FROM profiles p 
      JOIN deposits d ON d.user_id = p.user_id 
      WHERE p.user_id = ANY(level2_ids) AND d.status = 'completed' AND d.amount > 0
    ),
    'cEnthusiasts', 0,
    'totalOrders', (SELECT COUNT(*) FROM orders WHERE user_id = p_user_id),
    'boughtCount', (SELECT COUNT(*) FROM orders WHERE user_id = p_user_id AND status = 'completed'),
    'soldCount', (SELECT COUNT(*) FROM orders WHERE user_id = p_user_id AND status = 'sold')
  );
END;
$$ LANGUAGE plpgsql;
