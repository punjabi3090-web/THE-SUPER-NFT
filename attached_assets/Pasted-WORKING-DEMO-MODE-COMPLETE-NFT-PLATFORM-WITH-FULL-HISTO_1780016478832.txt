TASK 7: AUTO LOGGING SYSTEM

function addToHistory(type, data) {
  let history = JSON.parse(localStorage.getItem('myActivityHistory') || '[]');
  history.unshift({
    id: Date.now(),
    type: type,
    date: new Date().toLocaleString('en-US'),
  ...data
  });
  localStorage.setItem('myActivityHistory', JSON.stringify(history));
  updateHistoryUI();
}

AUTO TRIGGERS:
- Deposit complete → addToHistory('deposit', {...}) + addToHistory('reward', {title: 'Deposit Bonus'})
- Withdrawal complete → addToHistory('withdrawal', {...})
- Address Bind → addToHistory('security', {...})
- Google Auth → addToHistory('security', {...})
- NFT Reserve → addToHistory('reserve', {...})
- NFT Sell → addToHistory('sell', {...})
- Referral Join → addToHistory('reward', {title: 'Referral Bonus'})
- Airdrop → addToHistory('reward', {title: 'Airdrop Received'})
- Level Up → addToHistory('reward', {title: 'Level Up Reward'})
- Team Commission → addToHistory('commission', {...})
