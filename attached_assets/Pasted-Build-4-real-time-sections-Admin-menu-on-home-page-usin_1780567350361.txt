=== RLS POLICIES ===
ALTER TABLE deposits ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users view own deposits" ON deposits FOR SELECT USING (auth.uid() = user_id);

ALTER TABLE withdraws ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users view own withdraws" ON withdraws FOR SELECT USING (auth.uid() = user_id);

ALTER TABLE users ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users view referrals" ON users FOR SELECT USING (auth.uid() = id OR auth.uid() = referred_by);

ALTER TABLE admins ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Only admins access" ON admins FOR SELECT USING (auth.jwt() ->> 'email' = email);

=== DATABASE SETUP ===
ALTER TABLE users ADD COLUMN IF NOT EXISTS referred_by UUID REFERENCES auth.users(id);
ALTER TABLE deposits ADD COLUMN IF NOT EXISTS referred_by UUID;

  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email TEXT UNIQUE NOT NULL,
